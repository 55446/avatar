
famine = {
	enable = { always = yes }

	remove_trigger = {
		NOT = {
			check_variable = { famine_check < -0.5 }
			check_variable = { ration_variable = 0 }
		}
	}
	MONTHLY_POPULATION = -2
	stability_factor = -1
}
food_shortages = {
	enable = { always = yes }
	remove_trigger = {
		OR = {
			NOT = {
				check_variable = { ration_variable = 0 }
			}
			check_variable = { ration_penalty < -0.5 }
		}
	}
	MONTHLY_POPULATION = ration_penalty
	stability_factor = ration_penalty
}
food_surplus_idea = {
	enable = { always = yes }
	remove_trigger = {
		OR = {
			check_variable = { ration_variable = 0 }
			check_variable = { ration_variable = ration_max }
		}
	}
	MONTHLY_POPULATION = ration_buff
	stability_factor = ration_buff
}
burgerking = {
	enable = { always = yes }
	remove_trigger = {
		NOT = {
			check_variable = { ration_variable = ration_max }
		}
	}
	MONTHLY_POPULATION = 1
	stability_factor = 1
}

extreme_poverty = {
	enable = { always = yes }
	remove_trigger = {
		NOT = {
			check_variable = { consumer_and_luxury_goods_supply < half_consumer_demand }
		}
	}
	research_speed_factor = -0.3
}
poverty = {
	enable = { always = yes }
	remove_trigger = {
		NOT = {
			AND = {
				check_variable = { consumer_and_luxury_goods_supply < consumer_and_luxury_good_demand }
				check_variable = { consumer_and_luxury_goods_supply > half_consumer_demand }
			}
		}
	}
	research_speed_factor = -0.15
}
wealth = {
	enable = { always = yes }
	remove_trigger = {
		NOT = {
			AND = {
				check_variable = { consumer_and_luxury_goods_supply < double_consumer_demand }
				check_variable = { consumer_and_luxury_goods_supply > half_more_consumer_demand }
			}
		}
	}
	research_speed_factor = 0.1
}
extreme_wealth = {
	enable = { always = yes }
	remove_trigger = {
		NOT = {
			check_variable = { consumer_and_luxury_goods_supply > double_consumer_demand }
		}
	}
	research_speed_factor = 0.2
}
