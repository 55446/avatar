
famine = {
	enable = { always = yes }

	remove_trigger = {
		NOT = {
			check_variable = { ration_balance < 0 }
			check_variable = { ration_variable < ration_balance_int }
		}
	}
	MONTHLY_POPULATION = -2
	stability_factor = -1
}
food_shortages = {
	enable = { always = yes }
	remove_trigger = {
		NOT = {
			check_variable = { ration_balance < 0 }
			check_variable = { ration_variable < ration_balance_int }
		}
	}
	MONTHLY_POPULATION = -0.5
	stability_factor = -0.5
}
food_surplus_idea = {
	enable = { always = yes }
	remove_trigger = {
		NOT = {
			AND = {
				check_variable = { food_supply < double_food_demand }
				check_variable = { food_supply > half_more_food_demand }
			}
		}
	}
	MONTHLY_POPULATION = 0.25
	stability_factor = 0.25
}
