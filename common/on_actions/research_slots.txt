on_actions = {

	on_startup = {
		effect = {
			every_country = {
				

				#### slot from literacy

				if = {
					limit = {
						NOT = {
							has_country_flag = has_literacy_tech_slot
						}
						check_variable = { literacy_variable > 75 }
					}
					add_research_slot = 1
					set_country_flag = has_literacy_tech_slot
				}
				else_if = {
					limit = {
						has_country_flag = has_literacy_tech_slot
						NOT = {
							check_variable = { literacy_variable > 75 }
						}
					}
					add_research_slot = -1
					clr_country_flag = has_literacy_tech_slot
				}




				#### slot from economy
				
				if = {
					limit = {
						NOT = {
							has_country_flag = has_economy_tech_slot
						}
						check_variable = { num_of_civilian_factories > 50 }
						check_variable = { standard_of_living > 50 }
					}
					add_research_slot = 1
					set_country_flag = has_economy_tech_slot
				}
				else_if = {
					limit = {
						has_country_flag = has_economy_tech_slot
						NOT = {
							check_variable = { num_of_civilian_factories > 50 }
							check_variable = { standard_of_living > 50 }
						}
					}
					add_research_slot = -1
					clr_country_flag = has_economy_tech_slot
				}


				#### slot from international relations
				
				if = {
					limit = {
						NOT = {
							has_country_flag = has_relation_tech_slot
						}
						any_country = {
							has_opinion = {
								target = ROOT
								value > 80
							}
						}
					}
					add_research_slot = 1
					set_country_flag = has_relation_tech_slot
				}
				else_if = {
					limit = {
						has_country_flag = has_relation_tech_slot
						NOT = {
							any_country = {
								has_opinion = {
									target = ROOT
									value > 80
								}
							}
						}
					}
					add_research_slot = -1
					clr_country_flag = has_relation_tech_slot
				}
			}

			
		}
	}

	on_monthly = {

					check_variable = { standard_of_living > 50 }
				}
				add_research_slot = 1
				set_country_flag = has_economy_tech_slot
			}
			else_if = {
				limit = {
					has_country_flag = has_economy_tech_slot
					NOT = {
						check_variable = { num_of_civilian_factories > 50 }
						check_variable = { standard_of_living > 50 }
					}
				}
				add_research_slot = -1
				clr_country_flag = has_economy_tech_slot
			}


			#### slot from international relations
			if = {
				limit = {
					NOT = {
						has_country_flag = has_relation_tech_slot
					}
					any_country = {
						has_opinion = {
							target = ROOT
							value > 80
						}
					}
				}
				add_research_slot = 1
				set_country_flag = has_relation_tech_slot
			}
			else_if = {
				limit = {
					has_country_flag = has_relation_tech_slot
					NOT = {
						any_country = {
							has_opinion = {
								target = ROOT
								value > 80
							}
						}
					}
				}
				add_research_slot = -1
				clr_country_flag = has_relation_tech_slot
			}
			

		}
	}
